from pyscript import Element

def generate_link(event=None):
    url = Element("url-input").value.strip()
    output = Element("output")

    if not url:
        output.write("Please enter a URL.")
        return

    result = f"[{url}]({url})"
    output.write(result)

# Bind button click
Element("generate-btn").element.onclick = generate_link
